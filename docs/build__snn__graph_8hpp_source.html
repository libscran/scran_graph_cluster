130 // Reverse mapping is not parallel-frendly, so we don't construct this with the neighbor search.
131 std::vector<std::vector<Node_> > simple_hosts;
132 std::vector<std::vector<std::pair<Node_, Weight_> > > ranked_hosts;
133
134 if (options.weighting_scheme == Scheme::RANKED) {
135 ranked_hosts.resize(num_cells);
136 for (size_t i = 0; i < num_cells; ++i) {
137 ranked_hosts[i].emplace_back(i, 0); // each point is its own 0-th nearest neighbor
138 const auto& current = get_neighbors(i);
139 Weight_ rank = 1;
140 for (auto x : current) {
141 ranked_hosts[get_index(x)].emplace_back(i, rank);
142 ++rank;
143 }
144 }
145 } else {
146 simple_hosts.resize(num_cells);
147 for (size_t i = 0; i < num_cells; ++i) {
148 simple_hosts[i].emplace_back(i); // each point is its own 0-th nearest neighbor
149 const auto& current = get_neighbors(i);
150 for (auto x : current) {
151 simple_hosts[get_index(x)].emplace_back(i);
152 }
153 }
154 }
155
156 // Constructing the shared neighbor graph.
157 std::vector<std::vector<Node_> > edge_stores(num_cells);
158 std::vector<std::vector<Weight_> > weight_stores(num_cells);
159

161#ifdef _OPENMP
162 #pragma omp parallel num_threads(nthreads)
163#endif
164 {
165#else
166 SCRAN_CUSTOM_PARALLEL([&](size_t, size_t start, size_t length) -> void {
167#endif
168
169 std::vector<Weight_> current_score(num_cells);
170 std::vector<Node_> current_added;
171 current_added.reserve(num_cells);
172
173#ifndef SCRAN_CUSTOM_PARALLEL
174#ifdef _OPENMP
175 #pragma omp for
176#endif
177 for (size_t j = 0; j < num_cells; ++j) {
178#else
179 for (size_t j = start, end = start + length; j < end; ++j) {
180#endif
181 const Node_ j_cast = j;
182
183 const auto& current_neighbors = get_neighbors(j);
184 const size_t nneighbors = current_neighbors.size();
185
186 for (size_t i = 0; i <= nneighbors; ++i) {
187 // First iteration treats 'j' as the zero-th neighbor.
188 // Remaining iterations go through the neighbors of 'j'.
189 const Node_ cur_neighbor = (i == 0 ? j : get_index(current_neighbors[i-1]));
190
191 // Going through all observations 'h' for which 'cur_neighbor'
192 // is a nearest neighbor, a.k.a., 'cur_neighbor' is a shared
193 // neighbor of both 'h' and 'j'.
194 if (options.weighting_scheme == Scheme::RANKED) {
195 Weight_ i_cast = i;
196 for (const auto& h : ranked_hosts[cur_neighbor]) {
197 auto othernode = h.first;
198 if (othernode < j_cast) { // avoid duplicates from symmetry in the SNN calculations.
199 auto& existing_other = current_score[othernode];
200
201 // Recording the lowest combined rank per neighbor (casting to avoid overflow on Node_).
202 Weight_ currank = h.second + i_cast;
203 if (existing_other == 0) {
204 existing_other = currank;
205 current_added.push_back(othernode);
206 } else if (existing_other > currank) {
207 existing_other = currank;
208 }
209 }
210 }
211
212 } else {
213 for (const auto& othernode : simple_hosts[cur_neighbor]) {
214 if (othernode < j_cast) { // avoid duplicates from symmetry in the SNN calculations.
215 auto& existing_other = current_score[othernode];
216
217 // Recording the number of shared neighbors.
218 if (existing_other == 0) {
219 current_added.push_back(othernode);
220 }
221 ++existing_other;
222 }
223 }
224 }
225 }
226
227 // Converting to edges.
228 auto& current_edges = edge_stores[j];
229 current_edges.reserve(current_added.size() * 2);
230 auto& current_weights = weight_stores[j];
231 current_weights.reserve(current_added.size());
232
233 for (auto othernode : current_added) {
234 Weight_& otherscore = current_score[othernode];
235 Weight_ finalscore;
236 if (options.weighting_scheme == Scheme::RANKED) {
237 finalscore = static_cast<Weight_>(nneighbors) - 0.5 * static_cast<Weight_>(otherscore);
238 } else {
239 finalscore = otherscore;
240 if (options.weighting_scheme == Scheme::JACCARD) {
241 finalscore = finalscore / (2 * (nneighbors + 1) - finalscore);
242 }
243 }
244
245 current_edges.push_back(j);
246 current_edges.push_back(othernode);
247 current_weights.push_back(std::max(finalscore, 1e-6)); // Ensuring that an edge with a positive weight is always reported.
248
249 // Resetting all those added to zero.
250 otherscore = 0;
251 }
252 current_added.clear();
253
254#ifndef SCRAN_CUSTOM_PARALLEL
255 }
256 }
257#else
258 }
259 }, num_cells, options.num_threads);
260#endif
261
262 // Collating the total number of edges.
263 size_t nedges = 0;
264 for (const auto& w : weight_stores) {
265 nedges += w.size();
266 }
267
268 output.num_cells = num_cells;
269 output.weights.clear();
270 output.weights.reserve(nedges);
271 for (const auto& w : weight_stores) {
272 output.weights.insert(output.weights.end(), w.begin(), w.end());
273 }
274 weight_stores.clear();
275 weight_stores.shrink_to_fit(); // forcibly release memory so that we have some more space for edges.
276
277 output.edges.clear();
278 output.edges.reserve(nedges * 2);
279 for (const auto& e : edge_stores) {
280 output.edges.insert(output.edges.end(), e.begin(), e.end());
281 }
282
283 return;
