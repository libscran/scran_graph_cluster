117 // Reverse mapping is not parallel-frendly, so we don't construct this with the neighbor search.
118 std::vector<std::vector<Node_> > simple_hosts;
119 std::vector<std::vector<std::pair<Node_, Weight_> > > ranked_hosts;
120
121 if (options.weighting_scheme == Scheme::RANKED) {
122 ranked_hosts.resize(num_cells);
123 for (size_t i = 0; i < num_cells; ++i) {
124 ranked_hosts[i].emplace_back(i, 0); // each point is its own 0-th nearest neighbor
125 const auto& current = get_neighbors(i);
126 Weight_ rank = 1;
127 for (auto x : current) {
128 ranked_hosts[get_index(x)].emplace_back(i, rank);
129 ++rank;
130 }
131 }
132 } else {
133 simple_hosts.resize(num_cells);
134 for (size_t i = 0; i < num_cells; ++i) {
135 simple_hosts[i].emplace_back(i); // each point is its own 0-th nearest neighbor
136 const auto& current = get_neighbors(i);
137 for (auto x : current) {
138 simple_hosts[get_index(x)].emplace_back(i);
139 }
140 }
141 }
142
143 // Constructing the shared neighbor graph.
144 std::vector<std::vector<Node_> > edge_stores(num_cells);
145 std::vector<std::vector<Weight_> > weight_stores(num_cells);
146
147#ifndef SCRAN_CUSTOM_PARALLEL
148#ifdef _OPENMP
149 #pragma omp parallel num_threads(nthreads)
150#endif
151 {
152#else
153 SCRAN_CUSTOM_PARALLEL([&](size_t, size_t start, size_t length) -> void {
154#endif
155
156 std::vector<Weight_> current_score(num_cells);
157 std::vector<Node_> current_added;
158 current_added.reserve(num_cells);

160#ifndef SCRAN_CUSTOM_PARALLEL
161#ifdef _OPENMP
162 #pragma omp for
163#endif
164 for (size_t j = 0; j < num_cells; ++j) {
165#else
166 for (size_t j = start, end = start + length; j < end; ++j) {
167#endif
168 const Node_ j_cast = j;
169
170 const auto& current_neighbors = get_neighbors(j);
171 const size_t nneighbors = current_neighbors.size();
172
173 for (size_t i = 0; i <= nneighbors; ++i) {
174 // First iteration treats 'j' as the zero-th neighbor.
175 // Remaining iterations go through the neighbors of 'j'.
176 const Node_ cur_neighbor = (i == 0 ? j : get_index(current_neighbors[i-1]));
177
178 // Going through all observations 'h' for which 'cur_neighbor'
179 // is a nearest neighbor, a.k.a., 'cur_neighbor' is a shared
180 // neighbor of both 'h' and 'j'.
181 if (options.weighting_scheme == Scheme::RANKED) {
182 Weight_ i_cast = i;
183 for (const auto& h : ranked_hosts[cur_neighbor]) {
184 auto othernode = h.first;
185 if (othernode < j_cast) { // avoid duplicates from symmetry in the SNN calculations.
186 auto& existing_other = current_score[othernode];
187
188 // Recording the lowest combined rank per neighbor (casting to avoid overflow on Node_).
189 Weight_ currank = h.second + i_cast;
190 if (existing_other == 0) {
191 existing_other = currank;
192 current_added.push_back(othernode);
193 } else if (existing_other > currank) {
194 existing_other = currank;
195 }
196 }
197 }
198
199 } else {
200 for (const auto& othernode : simple_hosts[cur_neighbor]) {
201 if (othernode < j_cast) { // avoid duplicates from symmetry in the SNN calculations.
202 auto& existing_other = current_score[othernode];
203
204 // Recording the number of shared neighbors.
205 if (existing_other == 0) {
206 current_added.push_back(othernode);
207 }
208 ++existing_other;
209 }
210 }
211 }
212 }
213
214 // Converting to edges.
215 auto& current_edges = edge_stores[j];
216 current_edges.reserve(current_added.size() * 2);
217 auto& current_weights = weight_stores[j];
218 current_weights.reserve(current_added.size());
219
220 for (auto othernode : current_added) {
221 Weight_& otherscore = current_score[othernode];
222 Weight_ finalscore;
223 if (options.weighting_scheme == Scheme::RANKED) {
224 finalscore = static_cast<Weight_>(nneighbors) - 0.5 * static_cast<Weight_>(otherscore);
225 } else {
226 finalscore = otherscore;
227 if (options.weighting_scheme == Scheme::JACCARD) {
228 finalscore = finalscore / (2 * (nneighbors + 1) - finalscore);
229 }
230 }
231
232 current_edges.push_back(j);
233 current_edges.push_back(othernode);
234 current_weights.push_back(std::max(finalscore, 1e-6)); // Ensuring that an edge with a positive weight is always reported.
235
236 // Resetting all those added to zero.
237 otherscore = 0;
238 }
239 current_added.clear();
240
241#ifndef SCRAN_CUSTOM_PARALLEL
242 }
243 }
244#else
245 }
246 }, num_cells, options.num_threads);
247#endif
248
249 // Collating the total number of edges.
250 size_t nedges = 0;
251 for (const auto& w : weight_stores) {
252 nedges += w.size();
253 }
254
255 output.num_cells = num_cells;
256 output.weights.clear();
257 output.weights.reserve(nedges);
258 for (const auto& w : weight_stores) {
259 output.weights.insert(output.weights.end(), w.begin(), w.end());
260 }
261 weight_stores.clear();
262 weight_stores.shrink_to_fit(); // forcibly release memory so that we have some more space for edges.
263
264 output.edges.clear();
265 output.edges.reserve(nedges * 2);
266 for (const auto& e : edge_stores) {
267 output.edges.insert(output.edges.end(), e.begin(), e.end());
268 }
269
270 return;
