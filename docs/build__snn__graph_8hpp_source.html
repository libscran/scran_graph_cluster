132void build_snn_graph(size_t num_cells, GetNeighbors_ get_neighbors, GetIndex_ get_index, const BuildSnnGraphOptions& options, BuildSnnGraphResults<Node_, Weight_>& output) {
133 // Reverse mapping is not parallel-frendly, so we don't construct this with the neighbor search.
134 std::vector<std::vector<Node_> > simple_hosts;
135 std::vector<std::vector<std::pair<Node_, Weight_> > > ranked_hosts;
136
137 if (options.weighting_scheme == SnnWeightScheme::RANKED) {
138 ranked_hosts.resize(num_cells);
139 for (size_t i = 0; i < num_cells; ++i) {
140 ranked_hosts[i].emplace_back(i, 0); // each point is its own 0-th nearest neighbor
141 const auto& current = get_neighbors(i);
142 Weight_ rank = 1;
143 for (auto x : current) {
144 ranked_hosts[get_index(x)].emplace_back(i, rank);
145 ++rank;
146 }
147 }
148 } else {
149 simple_hosts.resize(num_cells);
150 for (size_t i = 0; i < num_cells; ++i) {
151 simple_hosts[i].emplace_back(i); // each point is its own 0-th nearest neighbor
152 const auto& current = get_neighbors(i);
153 for (auto x : current) {
154 simple_hosts[get_index(x)].emplace_back(i);
155 }
156 }
157 }
158
159 // Constructing the shared neighbor graph.
160 std::vector<std::vector<Node_> > edge_stores(num_cells);
161 std::vector<std::vector<Weight_> > weight_stores(num_cells);
162
163#ifndef SCRAN_CUSTOM_PARALLEL
164#ifdef _OPENMP
165 #pragma omp parallel num_threads(options.num_threads)
166#endif
167 {
168#else
169 SCRAN_CUSTOM_PARALLEL([&](size_t, size_t start, size_t length) -> void {
170#endif
171
172 std::vector<Weight_> current_score(num_cells);
173 std::vector<Node_> current_added;
174 current_added.reserve(num_cells);
175
176#ifndef SCRAN_CUSTOM_PARALLEL
177#ifdef _OPENMP
178 #pragma omp for
179#endif
180 for (size_t j = 0; j < num_cells; ++j) {
181#else
182 for (size_t j = start, end = start + length; j < end; ++j) {
183#endif
184 const Node_ j_cast = j;
185
186 const auto& current_neighbors = get_neighbors(j);
187 const size_t nneighbors = current_neighbors.size();
188
189 for (size_t i = 0; i <= nneighbors; ++i) {
190 // First iteration treats 'j' as the zero-th neighbor.
191 // Remaining iterations go through the neighbors of 'j'.
192 const Node_ cur_neighbor = (i == 0 ? j : get_index(current_neighbors[i-1]));
193
194 // Going through all observations 'h' for which 'cur_neighbor'
195 // is a nearest neighbor, a.k.a., 'cur_neighbor' is a shared
196 // neighbor of both 'h' and 'j'.
197 if (options.weighting_scheme == SnnWeightScheme::RANKED) {
198 Weight_ i_cast = i;
199 for (const auto& h : ranked_hosts[cur_neighbor]) {
200 auto othernode = h.first;
201 if (othernode < j_cast) { // avoid duplicates from symmetry in the SNN calculations.
202 auto& existing_other = current_score[othernode];
203
204 // Recording the lowest combined rank per neighbor (casting to avoid overflow on Node_).
205 Weight_ currank = h.second + i_cast;
206 if (existing_other == 0) {
207 existing_other = currank;
208 current_added.push_back(othernode);
209 } else if (existing_other > currank) {
210 existing_other = currank;
211 }
212 }
213 }
214
215 } else {
216 for (const auto& othernode : simple_hosts[cur_neighbor]) {
217 if (othernode < j_cast) { // avoid duplicates from symmetry in the SNN calculations.
218 auto& existing_other = current_score[othernode];
219
220 // Recording the number of shared neighbors.
221 if (existing_other == 0) {
222 current_added.push_back(othernode);
223 }
224 ++existing_other;
225 }
226 }
227 }
228 }
229
230 // Converting to edges.
231 auto& current_edges = edge_stores[j];
232 current_edges.reserve(current_added.size() * 2);
233 auto& current_weights = weight_stores[j];
234 current_weights.reserve(current_added.size());
235
236 for (auto othernode : current_added) {
237 Weight_& otherscore = current_score[othernode];
238 Weight_ finalscore;
239 if (options.weighting_scheme == SnnWeightScheme::RANKED) {
240 finalscore = static_cast<Weight_>(nneighbors) - 0.5 * static_cast<Weight_>(otherscore);
241 } else {
242 finalscore = otherscore;
243 if (options.weighting_scheme == SnnWeightScheme::JACCARD) {
244 finalscore = finalscore / (2 * (nneighbors + 1) - finalscore);
245 }
246 }
247
248 current_edges.push_back(j);
249 current_edges.push_back(othernode);
250 current_weights.push_back(std::max(finalscore, 1e-6)); // Ensuring that an edge with a positive weight is always reported.
251
252 // Resetting all those added to zero.
253 otherscore = 0;
254 }
255 current_added.clear();
256
257#ifndef SCRAN_CUSTOM_PARALLEL
258 }
259 }
260#else
261 }
262 }, num_cells, options.num_threads);
263#endif
264
265 // Collating the total number of edges.
266 size_t nedges = 0;
267 for (const auto& w : weight_stores) {
268 nedges += w.size();
269 }
270
271 output.num_cells = num_cells;
272 output.weights.clear();
273 output.weights.reserve(nedges);
274 for (const auto& w : weight_stores) {
275 output.weights.insert(output.weights.end(), w.begin(), w.end());
276 }
277 weight_stores.clear();
278 weight_stores.shrink_to_fit(); // forcibly release memory so that we have some more space for edges.
279
280 output.edges.clear();
281 output.edges.reserve(nedges * 2);
282 for (const auto& e : edge_stores) {
283 output.edges.insert(output.edges.end(), e.begin(), e.end());
284 }
285
286 return;
287}
